import csv
import os
import random
import string

def get_random_code():
    return random.choice(string.ascii_lowercase) + ''.join(random.choices(string.digits, k=3))

clips = [
    # Data removed for brevity
]

file_path = r"P:\AI\The Viral\The Girl\Nong Fah\fah_first_5_clips.tsv"

with open(file_path, 'w', newline='', encoding='utf-8') as f:
    writer = csv.writer(f, delimiter='\t')
    writer.writerow(['ID', 'หัวข้อ (Topic)', 'ฉาก (Scene)', 'Image Code', 'Image Prompt (สร้างแค่ครั้งเดียวต่อหัวข้อ)', 'Image Seedream', 'Video Prompt with บทพูด (Script)'])
    
    for clip in clips:
        img_code = get_random_code()
        
        # Format the descriptions
        easter_egg_str = f" {clip['easter_egg']}" if clip['easter_egg'] else ""
        
        # 1. Base Image Prompt
        base_img_prompt = f"[{clip['id']}-01] Authentic smartphone photo of a 28yo rural Thai girl. Face MUST look exactly like reference, IGNORE reference outfit/background. Highly detailed sharp face, clear flawless skin with minimal blemishes, light natural makeup, yet retaining realistic natural skin pores and micro-texture. Natural normal figure, soft and approachable, not skinny. She possesses a natural Cup D bust with a soft teardrop shape and realistic droop. Wearing a {clip['outfit']}. {clip['hair']}. She is in {clip['bg']}.{easter_egg_str} Natural ambient lighting, highly realistic, zero studio lighting, everyday lifestyle aesthetic."
        base_img_prompt = ' '.join(base_img_prompt.split())
        
        # 2. SD Image Prompt
        sd_img_prompt = f"[{clip['id']}-01SD] Authentic smartphone photo of a 28yo rural Thai girl. Face MUST look exactly like reference, IGNORE reference outfit/background. Highly detailed sharp face, clear flawless skin, retaining natural skin pores and micro-texture. Soft body shape with a normal waistline, slightly fleshy and curvy but not fat, soft and approachable. Large, prominent and heavy Cup D bust with a soft teardrop shape and realistic droop. Pushed up chest, emphasizing deep cleavage. Wearing a {clip['outfit']}. {clip['hair']}. She is in {clip['bg']}.{easter_egg_str} Natural ambient lighting, highly realistic, zero studio lighting, everyday lifestyle aesthetic."
        sd_img_prompt = ' '.join(sd_img_prompt.split())
        
        for sc in clip['scenes']:
            sc_num = sc['scene']
            
            # Setup Image columns logic
            final_img = base_img_prompt if sc_num == "01" else "(ใช้ภาพจาก Scene 01)"
            final_sd = sd_img_prompt if sc_num == "01" else "(ใช้ภาพจาก Scene 01)"
            
            # 3. Video Prompt
            # No newlines, no clothing description, correct nationality filter
            vid_prompt = f"[{clip['id']}-{sc_num}] Veo 3.1 Prompt (Girlfriend Voice): \"Authentic smartphone video, candid style. {sc['camera']} {sc['action']} The young woman speaks directly to the camera with a soft, sweet, flirty, youthful voice, saying: '{sc['script']}' She speaks slowly and warmly. [Audio: Raw unedited speech, natural ambient sounds only, zero music]\""
            vid_prompt = ' '.join(vid_prompt.split())
            
            writer.writerow([
                clip['id'], 
                clip['topic'], 
                sc_num, 
                img_code, 
                final_img, 
                final_sd, 
                vid_prompt
            ])

print(f"Successfully generated {file_path}")
