### Core Implementation

```javascript
// scripts/hydra/inspector.js  (cron: 0 */6 * * *)
import Database from 'better-sqlite3';

async function inspect() {
  const db = new Database('/root/brain-app/brain.db');
  const findings = [];
  
  // 1. Performance Drift Detection (pure SQL)
  const cronStats = db.prepare(`
    SELECT script_id, 
           AVG(duration_ms) as avg_ms,
           AVG(CASE WHEN start_time > datetime('now','-7 days') 
                THEN duration_ms END) as recent_avg_ms
    FROM cc_cron_run_logs WHERE status = 'success'
    GROUP BY script_id
  `).all();
  
  for (const stat of cronStats) {
    if (stat.recent_avg_ms > stat.avg_ms * 2) {
      findings.push({
        category: 'performance_drift', severity: 'warning',
        title: `${stat.script_id} is ${(stat.recent_avg_ms/stat.avg_ms).toFixed(1)}x slower`,
        evidence: { baseline: stat.avg_ms, recent: stat.recent_avg_ms },
        affected_component: stat.script_id
      });
    }
  }

  // 2. Recurring Error Pattern
  const errorPatterns = db.prepare(`
    SELECT check_id, message, COUNT(*) as cnt
    FROM watchdog_logs WHERE status='FAIL' AND timestamp > datetime('now','-3 days')
    GROUP BY check_id, message HAVING cnt >= 3
  `).all();
  for (const err of errorPatterns) {
    findings.push({
      category: 'recurring_error', severity: 'critical',
      title: `Recurring: ${err.check_id} (${err.cnt}x in 3 days)`
    });
  }

  // 3. Knowledge Staleness
  const stale = db.prepare(`
    SELECT COUNT(*) as cnt FROM knowledge_entries  
    WHERE updated_at < datetime('now','-30 days')
    AND (valid_to IS NULL OR valid_to > date('now'))
  `).get();
  if (stale.cnt > 10) {
    findings.push({ category: 'knowledge_staleness', severity: 'info',
      title: `${stale.cnt} knowledge entries stale (30+ days)` });
  }

  // 4. Growth Goals Check
  // ... compare current metrics vs goals in evolution_goals table

  // 5. Health Score
  const criticals = findings.filter(f => f.severity === 'critical').length;
  const warnings = findings.filter(f => f.severity === 'warning').length;
  const healthScore = Math.max(0, 100 - (criticals * 15) - (warnings * 5));

  // Save to DB (Ideator will read this)
  db.prepare(`
    INSERT INTO evolution_log (proposal_id, title, category, status, raw_data)
    VALUES (?, ?, ?, 'finding', ?)
  `).run(/* ... per finding */);

  // Call Gemini Flash for human-readable summary
  const summary = await callGeminiFlash(`สรุป inspection findings: ${JSON.stringify(findings)}`);
  
  // Discord alert if critical
  if (criticals > 0) await discordAlert(summary);

  db.close();
}
```

## 💡 Agent 2: IDEATOR

**Role:** "สมอง" รับ Inspection Report แล้วคิดวิเคราะห์ + เสนอแผน

**Model:**
- Finding เล็ก (performance, cost) → **Gemini Pro**
- Finding ใหญ่ (architecture, logic) → **Claude Sonnet (Thinking)**
- Innovation Mode (ไม่มี bug) → **Gemini Pro**

**Cost:** ~3,000-8,000 tokens / finding (รันแค่ตอนมี finding)

### กลไก 4 ขั้น

```
1. ROOT CAUSE ANALYZER  ← ดึง Skills + Knowledge Graph + Error Logs
2. SOLUTION GENERATOR   ← เสนอ 2-3 ทางเลือก rank ตาม Impact/Effort
3. INNOVATION SCANNER   ← ถ้าไม่มี bug → มองหาโอกาสใหม่
4. RISK ASSESSOR        ← classify: safe/medium/high ก่อนเสนอ
```

### แหล่งข้อมูลที่ Ideator ดึงมาวิเคราะห์

| Source | วิธีใช้ |

| `corrections.md` | สิ่งที่ User เคยแก้ไข → ไม่เสนอสิ่งที่ User ไม่ชอบ |

## 🧬 Agent 3: EVOLVER

**Role:** "มือ" รับ Proposal แล้วลงมือทำจริง (code, test, deploy)

**Model:**
- SAFE changes → **Claude Sonnet** (ดีพอ + ประหยัด)
- MEDIUM changes → **Claude Opus** (พลาดไม่ได้)
- HIGH risk → ไม่ให้ AI ทำคนเดียว → เสนอแผนอย่างเดียว

**Cost:** ~5,000-20,000 tokens / change (รันแค่ตอน proposal approved)

### Safety-First Risk Classification

| Risk Level | ตัวอย่าง | ใครอนุมัติ |
|---|---|---|
| 🟢 **Safe** | เพิ่ม cache, แก้ prompt, optimize query | **Auto-deploy ทันที** |
| 🟡 **Medium** | เปลี่ยน cron schedule, แก้ 2+ files | **Dry Run → รอ Human approve → AG implement** |
| 🔴 **High** | เปลี่ยน architecture, DB schema ใหม่ | **เสนอแผนเท่านั้น → คุย AG** |

### Execution Pipeline

```javascript
// scripts/hydra/evolver.js

async function executeProposal(proposal) {
  // 1. Safety Gate
  if (proposal.risk_level === 'high') {
    await notifyDiscord(`🔴 HIGH RISK — human decision needed: ${proposal.title}`);
    return { status: 'awaiting_human' };
  }

  // 2. Create Git Snapshot (rollback point)
  const snapshot = execSync('git log -1 --format="%H"').toString().trim();

  // 3. Generate Code Patch (Claude Sonnet/Opus → unified diff)
  const patch = await callClaude(buildCodePatchPrompt(proposal));

  // 4. Apply Patch
  await applyPatch(patch);

  // 5. Dry Run Validation
  const result = execSync(`node ${proposalFile} --dry-run`);
  if (result.exitCode !== 0) {
    await rollback(snapshot);
    await emitSkillFail(proposal, result);
    return { status: 'validation_failed' };
  }

  // 6. Deploy (Safe) or Stage (Medium)
  if (proposal.risk_level === 'safe') {
    execSync('git commit -m "evo: ..."');
    execSync('git push vps');
    execSync('pm2 restart brain-app');
    
    // 7. Monitor 10 minutes (auto-rollback if crash)
    await monitorPostDeploy(10 * 60_000, snapshot);
    
    // 8. Emit Skill (learning)
    await emitSkill(proposal);
    return { status: 'deployed' };
  } else {
    await notifyDiscord(`🟡 MEDIUM risk staged: ${proposal.title}\n!evo approve ${proposal.id}`);
    return { status: 'staged_for_review' };
  }
}

async function monitorPostDeploy(durationMs, snapshot) {
  for (let i = 0; i < durationMs / 60_000; i++) {
    await sleep(60_000);
    const health = await quickHealthCheck(); // API ping + PM2 status
    if (!health.ok) {
      await rollback(snapshot);
      await notifyDiscord(`🚨 Auto-rollback! Reason: ${health.reason}`);
      return;
    }
  }
}
```

## 🔑 Architecture Decisions & Alternatives Considered

### Decision 1: ทำไมไม่ใช้ CrewAI / LangGraph
- Python-only → ต้องเปลี่ยน stack ทั้งหมด (Node.js → Python)
- Overhead สูงมาก สำหรับ cron scripts ที่เป็น vanilla Node
- เราสร้าง lightweight orchestrator ของเราเองบน infra ที่มีอยู่ดีกว่า

### Decision 2: ทำไม Inspector ใช้ SQL 80% ไม่ใช้ AI
- ตรวจสอบตัวเลข = deterministic ไม่ต้องการ creativity
- SQL เร็วกว่า, ถูกกว่า, แม่นกว่า AI สำหรับ numeric checks
- AI ใช้แค่ตอนต้องการ "อธิบายผลให้คนเข้าใจ"

### Decision 3: Evolver "Report Only" ก่อน Auto-Deploy
- สัปดาห์แรก → แค่รายงานว่า "จะทำอะไร" ไม่ทำจริง
- ให้คนตรวจสอบ pattern ก่อนเชื่อใจ
- ค่อยๆ เปิด auto-deploy เฉพาะ safe changes

### Decision 4: Growth Goals vs Pure Maintenance
- ถ้ามีแค่ Inspector+Ideator+Evolver → วน Loop fix bugs อย่างเดียว
- Growth Goals ทำให้ Ideator เข้า "Innovation Mode" เมื่อไม่มี bug
- ทำให้ระบบ "โต" ไม่ใช่แค่ "ซ่อม"

### Alternative ที่ไม่เลือก
- ❌ ใช้ GitHub Actions แทน cron → ต้อง trigger by event ไม่ได้ autonomous
- ❌ ให้ AG เป็น Evolver ทั้งหมด → ต้องมีคนเปิด conversation ตลอด ไม่ autonomous
- ❌ ใช้ Docker containers สำหรับ meta-agents → overkill สำหรับ Node scripts
