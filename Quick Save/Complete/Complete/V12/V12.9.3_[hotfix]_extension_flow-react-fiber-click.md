# Fix God Flow Studio Send Button

## Context
ผู้ใช้พบปัญหา God Flow Studio (v3.6.0) พิมพ์ Prompt ได้แต่ **"Send button not clicking"** (ปุ่มไม่ยอมส่ง) เมื่อเทียบกับ `Flow-Auto-2026- V. 2.0` ที่ทำงานได้ปกติ
สาเหตุคือ Google Flow ใช้ React ซึ่งมีการดัก event แบบ Synthetic และมีการ disable ปุ่มเมื่อกำลังโหลด หรือตรวจสอบ Fiber ทำให้การ `.click()` ปกติไม่ทำงาน หรือทำงานไม่ได้ผล

## What we decided to do (ทำไมถึงเลือกแนวทางนี้)
1. **Traverse React Fiber (`onClick`)**: แทนที่จะ click ที่ DOM ตรงๆ เราต้องเจาะเข้าไปใน React Fiber (`__reactFiber$`) เพื่อจำลองเหตุการณ์ click ลงไปที่ `onClick` handler ของ React โดยตรง
2. **Execute in `MAIN` world**: Content script ทำงานใน `ISOLATED` world ไม่สามารถแตะ React Fiber ได้ เราจึงย้ายคำสั่งไปไว้ที่ `background.js` (ใช้ `chrome.scripting.executeScript({ world: 'MAIN' })`)
3. **Advanced Button Heuristics (`findSendButton`)**: ปุ่ม header/nav ชอบหลอกว่าเป็นปุ่ม send เราเลยเพิ่ม `Navigation Guard` เพื่อกันปุ่มเหล่านั้น (ตรวจสอบจาก `aria-label`, `top < 80`, และ CSS Context)

## Timeline
- **ตรวจสอบ Reference Extension**: ไปแงะดูที่ `O:\Program\My Extension\Flow-Auto-2026- V. 2.0 [25-5-2026]` พบว่ามีท่าลับ `MAIN_WORLD_CLICK`
- **แก้ไข `background.js`**: วางโค้ด `MAIN_WORLD_CLICK` ที่ใช้ MutationObserver กันปุ่มโดน React disable และยิง event ทุกรูปแบบลงไป
- **แก้ไข `content.js`**: อัพเกรด `findSendButton` เปลี่ยนไปให้ใช้ heuristics จาก Flow Auto 2026 (ดักทาง icon/text/CSS เต็มพิกัด)

function findSendButton() {
    function _isNavOrLink(b) {
        if (!b) return true;
        if (b.tagName === 'A') return true;
        if (b.closest('a[href]')) return true;
        if (b.closest('header') || b.closest('nav')) return true;
        if (b.closest('[role="banner"]') || b.closest('[role="navigation"]')) return true;
        const br = b.getBoundingClientRect();
        if (br.top < 80) return true;
        const al = (b.getAttribute('aria-label') || '').toLowerCase();
        const tl = (b.getAttribute('title') || '').toLowerCase();
        const tx = (b.textContent || '').trim().toLowerCase();
        const navKw = ['new project', 'new flow', 'create new', 'home', 'go back', 'project list', 'projects', 'log out', 'sign out', 'profile'];
        for (let nk = 0; nk < navKw.length; nk++) {
            if (al.indexOf(navKw[nk]) !== -1) return true;
            if (tl.indexOf(navKw[nk]) !== -1) return true;
            if (tx.length < 30 && tx.indexOf(navKw[nk]) !== -1) return true;
        }
        return false;
    }

    function findButtonByIcon(iconText, container) {
        const root = container || document;
        const buttons = root.querySelectorAll('button');
        for (const btn of buttons) {
            if (btn.offsetParent === null) continue; // skip hidden
            const icons = btn.querySelectorAll('i.google-symbols, i[class*="google-symbols"], i');
            for (const icon of icons) {
                const text = (icon.textContent || '').trim();
                if (text === iconText || text.startsWith(iconText)) {
                    return btn;
                }
            }
        }
        return null;
    }

    const slate = document.querySelector('[data-slate-editor="true"]');
    if (slate) {

        for (let up = 0; up < 8 && container; up++) {
            container = container.parentElement;
            if (!container) break;
            const btn = findButtonByIcon('arrow_forward', container);
            if (btn && !_isNavOrLink(btn)) return btn;
        }
    }

    // วิธีที่ 2: หาจาก icon "arrow_forward" ทั้งหน้า
    const globalBtn = findButtonByIcon('arrow_forward');
    if (globalBtn && !_isNavOrLink(globalBtn)) return globalBtn;

    // วิธีที่ 3: หาจาก hidden text "Create"/"Generate" ใน button
    const allBtns = document.querySelectorAll('button');
    for (const btn of allBtns) {
        if (btn.offsetParent === null) continue;
        if (_isNavOrLink(btn)) continue;
        const icon = btn.querySelector('i.google-symbols, i[class*="google-symbols"], i');
        const iconText = icon ? (icon.textContent || '').trim() : '';
        if (iconText !== 'arrow_forward') continue;
        const spans = btn.querySelectorAll('span');
        for (const span of spans) {
            const text = (span.textContent || '').trim();
            if (text === 'Create' || text === 'Generate') {
                return btn;
            }
        }

  // --- MAIN World Button Click (bypasses React isolated world restriction) ---
  if (message.type === 'MAIN_WORLD_CLICK') {
    const tabId = sender.tab?.id;
    const { markerAttr } = message;
    if (!tabId || !markerAttr) {
      sendResponse({ ok: false, error: 'Missing tabId or markerAttr' });
      return;
    }
    chrome.scripting.executeScript({
      target: { tabId },
      world: 'MAIN',
      func: function(attr) {
        try {
          const btn = document.querySelector('[' + attr + ']');
          if (!btn) return { ok: false, error: 'Button not found in MAIN world' };

          // MutationObserver to prevent React from setting disabled back
          var observer = new MutationObserver(function(mutations) {
            for (var m = 0; m < mutations.length; m++) {
              if (mutations[m].attributeName === 'disabled' || mutations[m].attributeName === 'aria-disabled') {
                btn.disabled = false;
                btn.removeAttribute('disabled');
                btn.removeAttribute('aria-disabled');
              }
            }
          });
          observer.observe(btn, { attributes: true, attributeFilter: ['disabled', 'aria-disabled'] });

          // Force-enable
          btn.disabled = false;
          btn.removeAttribute('disabled');
          btn.removeAttribute('aria-disabled');

          var methods = [];

          // React fiber onClick
          try {
            var fiberKey = Object.keys(btn).find(function(k) { return k.startsWith('__reactFiber$') || k.startsWith('__reactInternalInstance$'); });
            if (fiberKey) {
              var f = btn[fiberKey];
              for (var d = 0; d < 20 && f; d++) {
                if (f.memoizedProps && typeof f.memoizedProps.onClick === 'function') {
                  var rect = btn.getBoundingClientRect();
                  f.memoizedProps.onClick({ 
                    preventDefault: function(){}, 
                    stopPropagation: function(){}, 
                    nativeEvent: new MouseEvent('click', { bubbles: true, clientX: rect.left + rect.width/2, clientY: rect.top + rect.height/2 }), 
                    target: btn, 
                    currentTarget: btn, 
                    type: 'click', 
                    bubbles: true 
                  });
                  methods.push('fiberOnClick@' + d);
                  break;
                }
                f = f.return;
              }
            }
          } catch (e) {
            methods.push('fiberError:' + e.message);
          }

          // React __reactProps onClick
          try {
            var propsKey = Object.keys(btn).find(function(k) { return k.startsWith('__reactProps$'); });
            if (propsKey && btn[propsKey] && typeof btn[propsKey].onClick === 'function') {
              btn[propsKey].onClick({ preventDefault: function(){}, stopPropagation: function(){}, nativeEvent: new MouseEvent('click'), target: btn, currentTarget: btn });
              methods.push('propsOnClick');
            }
          } catch (e) {
            methods.push('propsError:' + e.message);
          }

          // Full pointer/mouse events
          try {
            var rect2 = btn.getBoundingClientRect();
            var cx = rect2.left + rect2.width / 2, cy = rect2.top + rect2.height / 2;
            btn.dispatchEvent(new PointerEvent('pointerdown', { bubbles: true, cancelable: true, clientX: cx, clientY: cy, pointerId: 1, pointerType: 'mouse', isPrimary: true }));
            btn.dispatchEvent(new MouseEvent('mousedown', { bubbles: true, cancelable: true, clientX: cx, clientY: cy, button: 0, buttons: 1 }));
            btn.dispatchEvent(new PointerEvent('pointerup', { bubbles: true, cancelable: true, clientX: cx, clientY: cy, pointerId: 1, pointerType: 'mouse', isPrimary: true }));
            btn.dispatchEvent(new MouseEvent('mouseup', { bubbles: true, cancelable: true, clientX: cx, clientY: cy, button: 0, buttons: 0 }));
            btn.dispatchEvent(new MouseEvent('click', { bubbles: true, cancelable: true, clientX: cx, clientY: cy, button: 0 }));
            btn.click();
            methods.push('mouseEvents');
          } catch (e) {
            methods.push('mouseEventsError:' + e.message);
          }

          // Submit form if exists
          try {
            var form = btn.closest('form');
            if (form) {
              form.requestSubmit ? form.requestSubmit(btn) : form.submit();
              methods.push('formSubmit');
            }
          } catch (e) {}

          // Disconnect observer after 2s
          setTimeout(function() { observer.disconnect(); }, 2000);

          return { ok: true, methods: methods };
        } catch (err) {
          return { ok: false, error: err.message };
        }
      },
      args: [markerAttr]
    }).then((results) => {
      const res = results && results[0] && results[0].result;
      sendResponse(res && res.ok ? { ok: true, methods: res.methods } : { ok: false, error: (res && res.error) || 'unknown' });
    }).catch((err) => {
      console.error('[Background] MAIN_WORLD_CLICK error:', err);
      sendResponse({ ok: false, error: err.message });
    });
    return true; // Async response
  }
